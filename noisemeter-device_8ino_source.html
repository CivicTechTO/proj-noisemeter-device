273 void loop() {
274  readMicrophoneData();
275 
276 #ifndef UPLOAD_DISABLED
277  const auto now = Timestamp();
278 
279  if (lastUpload.secondsBetween(now) >= UPLOAD_INTERVAL_SEC) {
280  packets.front().timestamp = now;
281 
282  if (WiFi.status() != WL_CONNECTED) {
283  SERIAL.println("Attempting WiFi reconnect...");
284  WiFi.reconnect();
285  delay(5000);
286  }
287 
288  if (WiFi.status() == WL_CONNECTED) {
289  {
290  std::optional<Blinker> bl;
291 
292  // Only blink if there's multiple packets to send
293  if (++packets.cbegin() != packets.cend())
294  bl.emplace(300);
295 
296  packets.remove_if([](const auto& pkt) {
297  if (pkt.count > 0) {
298  const auto payload = createJSONPayload(pkt);
299  return uploadData(payload) == 0;
300  } else {
301  return true; // Discard empty packets
302  }
303  });
304  }
305 
306 #if defined(BOARD_ESP32_PCB)
307  // We have WiFi: also check for software updates
308  if (lastOTACheck.secondsBetween(now) >= OTA_INTERVAL_SEC) {
309  lastOTACheck = now;
310  SERIAL.println("Checking for updates...");
311 
312  OTAUpdate ota (cert_ISRG_Root_X1);
313  if (ota.available()) {
314  SERIAL.print(ota.version);
315  SERIAL.println(" available!");
316 
317  if (ota.download()) {
318  SERIAL.println("Download success! Restarting...");
319  delay(1000);
320  ESP.restart();
321  } else {
322  SERIAL.println("Update download failed.");
323  }
324  } else {
325  SERIAL.println("No update available.");
326  }
327  }
328 #endif // BOARD_ESP32_PCB
329  }
330 
331  if (!packets.empty()) {
332  const auto count = std::distance(packets.cbegin(), packets.cend());
333  SERIAL.print(count);
334  SERIAL.println(" packets still need to be sent!");
335 
336  if (count >= MAX_SAVED_PACKETS) {
337  SERIAL.println("Discarded a packet!");
338  packets.pop_back();
339  }
340  }
341 
342  // Create new packet for next measurements
343  packets.emplace_front();
344  lastUpload = now;
345  }
346 #endif // !UPLOAD_DISABLED
347 }

371 void saveNetworkCreds(WebServer& httpServer) {
372  // Confirm that the form was actually submitted.
373  if (httpServer.hasArg("ssid") && httpServer.hasArg("psk")) {
374  const auto id = String(buildDeviceId());
375  const auto ssid = Secret::encrypt(id, httpServer.arg("ssid"));
376  const auto psk = Secret::encrypt(id, httpServer.arg("psk"));
377 
378  // Confirm that the given credentials will fit in the allocated EEPROM space.
379  if (!ssid.isEmpty() && Creds.canStore(ssid) && Creds.canStore(psk)) {
380  Creds.set(Storage::Entry::SSID, ssid);
381  Creds.set(Storage::Entry::Passkey, psk);
382  Creds.commit();
383 
384  ESP.restart(); // Software reset.
385  }
386  }
387 
388  // TODO inform user that something went wrong...
389  SERIAL.println("Error: Invalid network credentials!");
390 }

543 void readMicrophoneData() {
544  // Block and wait for microphone values from I2S
545  //
546  // Data is moved from DMA buffers to our 'samples' buffer by the driver ISR
547  // and when there is requested ammount of data, task is unblocked
548  //
549  // Note: i2s_read does not care it is writing in float[] buffer, it will write
550  // integer values to the given address, as received from the hardware peripheral.
551  size_t bytes_read;
552  i2s_read(I2S_PORT, samples, sizeof(samples), &bytes_read, portMAX_DELAY);
553 
554  // Convert (including shifting) integer microphone values to floats,
555  // using the same buffer (assumed sample size is same as size of float),
556  // to save a bit of memory
557  auto int_samples = reinterpret_cast<SAMPLE_T*>(samples);
558 
559  for (int i = 0; i < SAMPLES_SHORT; i++) samples[i] = MIC_CONVERT(int_samples[i]);
560 
561  // Apply equalization and calculate Z-weighted sum of squares,
562  // writes filtered samples back to the same buffer.
563  q.sum_sqr_SPL = MIC_EQUALIZER.filter(samples, samples, SAMPLES_SHORT);
564 
565  // Apply weighting and calucate weigthed sum of squares
566  q.sum_sqr_weighted = WEIGHTING.filter(samples, samples, SAMPLES_SHORT);
567 
568  // Calculate dB values relative to MIC_REF_AMPL and adjust for microphone reference
569  double short_RMS = sqrt(double(q.sum_sqr_SPL) / SAMPLES_SHORT);
570  double short_SPL_dB = MIC_OFFSET_DB + MIC_REF_DB + 20 * log10(short_RMS / MIC_REF_AMPL);
571 
572  // In case of acoustic overload or below noise floor measurement, report infinty Leq value
573  if (short_SPL_dB > MIC_OVERLOAD_DB) {
574  Leq_sum_sqr = MIC_OVERLOAD_DB;
575  } else if (isnan(short_SPL_dB) || (short_SPL_dB < MIC_NOISE_DB)) {
576  Leq_sum_sqr = MIC_NOISE_DB;
577  }
578 
579  // Accumulate Leq sum
580  Leq_sum_sqr += q.sum_sqr_weighted;
581  Leq_samples += SAMPLES_SHORT;
582 
583  // When we gather enough samples, calculate new Leq value
584  if (Leq_samples >= SAMPLE_RATE * LEQ_PERIOD) {
585  double Leq_RMS = sqrt(Leq_sum_sqr / Leq_samples);
586  Leq_dB = MIC_OFFSET_DB + MIC_REF_DB + 20 * log10(Leq_RMS / MIC_REF_AMPL);
587  Leq_sum_sqr = 0;
588  Leq_samples = 0;
589 
590  packets.front().add(Leq_dB);
591  printReadingToConsole(Leq_dB);
592  }
593 }
