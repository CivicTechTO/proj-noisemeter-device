57 int sos_filter_f32(float *input, float *output, int len, const SOS_Coefficients &coeffs, SOS_Delay_State &w) {
58  // Assumes a0 and b0 coefficients are one (1.0)
59  float f0 = coeffs.b1;
60  float f1 = coeffs.b2;
61  float f2 = coeffs.a1;
62  float f3 = coeffs.a2;
63  float f4 = w.w0;
64  float f5 = w.w1;
65  for (; len > 0; len--) {
66  float f6 = *input++;
67  f6 += f2 * f4; // coeffs.a1 * w0
68  f6 += f3 * f5; // coeffs.a2 * w1
69  float f7 = f6; // b0 assumed 1.0
70  f7 += f0 * f4; // coeffs.b1 * w0
71  f7 += f1 * f5; // coeffs.b2 * w1 -> result
72  *output++ = f7;
73  f5 = f4; // w1 = w0
74  f4 = f6; // w0 = f6
75  }
76  w.w0 = f4;
77  w.w1 = f5;
78  return 0;
79 };

91 float sos_filter_sum_sqr_f32(float *input, float *output, int len, const SOS_Coefficients &coeffs, SOS_Delay_State &w, float gain) {
92  // Assumes a0 and b0 coefficients are one (1.0)
93  float f0 = coeffs.b1;
94  float f1 = coeffs.b2;
95  float f2 = coeffs.a1;
96  float f3 = coeffs.a2;
97  float f4 = w.w0;
98  float f5 = w.w1;
99  float f6 = gain;
100  float sum_sqr = 0;
101  for (int i = len; i > 0; i--) {
102  float f7 = *input++;
103  f7 += f2 * f4; // coeffs.a1 * w0
104  f7 += f3 * f5; // coeffs.a2 * w1;
105  float f8 = f7; // b0 assumed 1.0
106  f8 += f0 * f4; // coeffs.b1 * w0;
107  f8 += f1 * f5; // coeffs.b2 * w1;
108  float f9 = f8 * f6; // f8 * gain -> result
109  *output++ = f9;
110  f5 = f4; // w1 = w0
111  f4 = f7; // w0 = f7;
112  sum_sqr += f9 * f9; // sum_sqr += f9 * f9;
113  }
114  w.w0 = f4;
115  w.w1 = f5;
116  return sum_sqr;
117 };

164  inline float filter(float *input, float *output, size_t len) {
165  if ((num_sos < 1) || (sos == NULL) || (w == NULL)) return 0;
166  float *source = input;
167  // Apply all but last Second-Order-Section
168  for (int i = 0; i < (num_sos - 1); i++) {
169  sos_filter_f32(source, output, len, sos[i], w[i]);
170  source = output;
171  }
172  // Apply last SOS with gain and return the sum of squares of all samples
173  return sos_filter_sum_sqr_f32(source, output, len, sos[num_sos - 1], w[num_sos - 1], gain);
174  }

204  inline float filter(float *input, float *output, size_t len) {
205  float sum_sqr = 0;
206  float s;
207  for (int i = 0; i < len; i++) {
208  s = input[i];
209  sum_sqr += s * s;
210  }
211  if (input != output) {
212  for (int i = 0; i < len; i++) output[i] = input[i];
213  }
214  return sum_sqr;
215  };
